    <script type="module">
        import { initializeApp } from 'https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js';
        import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from 'https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js';
        import { getFirestore, collection, query, onSnapshot, doc, setDoc, getDoc, updateDoc, addDoc, deleteDoc, serverTimestamp } from 'https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js';

        const { createApp, ref, computed, onMounted, watch } = Vue;

        const firebaseConfig = JSON.parse(__firebase_config);
        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'blue-island-analytics-v1';
        const apiKey = "";

        createApp({
            setup() {
                const currentTab = ref('dashboard');
                const messages = ref([]);
                const loading = ref(true);
                const simulating = ref(false);
                const lastSync = ref('--:--');
                const cloudWords = ref([]);
                let chartInstance = null;

                const aiConfig = ref({
                    systemPrompt: "你是岛屿守护灵，语气治愈且神秘。",
                    auditPrompt: "如果是违规内容（色情、暴力、仇恨言论），请输出具体的违规原因；如果是安全的，仅输出 'safe'。",
                    autoAudit: true
                });

                const sim = ref({ author: 'Explorer_' + Math.floor(Math.random()*99), content: '' });

                const tabName = computed(() => {
                    const map = { dashboard: '数据仪表盘', messages: '留言管理', ai: 'AI 核心配置', simulator: '模拟沙盒' };
                    return map[currentTab.value];
                });

                const violatedCount = computed(() => messages.value.filter(m => m.isViolated).length);
                const activeUsersCount = computed(() => new Set(messages.value.map(m => m.author)).size);

                onMounted(async () => {
                    const initAuth = async () => {
                        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                            await signInWithCustomToken(auth, __initial_auth_token);
                        } else {
                            await signInAnonymously(auth);
                        }
                    };
                    await initAuth();
                    
                    onAuthStateChanged(auth, u => {
                        if (u) {
                            fetchMessages();
                            fetchSettings();
                        }
                    });
                });

                const fetchMessages = () => {
                    const q = query(collection(db, 'artifacts', appId, 'public', 'data', 'messages'));
                    onSnapshot(q, (snap) => {
                        messages.value = snap.docs.map(d => ({ id: d.id, ...d.data() }))
                            .sort((a,b) => (b.createdAt?.seconds || 0) - (a.createdAt?.seconds || 0));
                        loading.value = false;
                        lastSync.value = new Date().toLocaleTimeString();
                        
                        if (currentTab.value === 'dashboard') {
                            generateCloud();
                            updateChart();
                        }
                    }, (err) => console.error(err));
                };

                const fetchSettings = async () => {
                    const snap = await getDoc(doc(db, 'artifacts', appId, 'public', 'data', 'settings', 'aiConfig'));
                    if (snap.exists()) aiConfig.value = { ...aiConfig.value, ...snap.data() };
                };

                const saveSettings = async () => {
                    loading.value = true;
                    await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'settings', 'aiConfig'), aiConfig.value);
                    loading.value = false;
                };

                const postSim = async () => {
                    if (!sim.value.content) return;
                    simulating.value = true;
                    let isViolated = false;
                    let auditReason = "";

                    if (aiConfig.value.autoAudit) {
                        try {
                            const res = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                                method: 'POST',
                                headers: { 'Content-Type': 'application/json' },
                                body: JSON.stringify({
                                    contents: [{ parts: [{ text: `审核规则: ${aiConfig.value.auditPrompt}\n内容: ${sim.value.content}` }] }]
                                })
                            });
                            const data = await res.json();
                            const resText = data.candidates?.[0]?.content?.parts?.[0]?.text?.trim() || "";
                            if (resText.toLowerCase() !== 'safe') {
                                isViolated = true;
                                auditReason = resText;
                            }
                        } catch (e) { console.error(e); }
                    }

                    await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'messages'), {
                        ...sim.value,
                        isViolated,
                        auditReason,
                        createdAt: serverTimestamp()
                    });

                    sim.value.content = '';
                    simulating.value = false;
                    currentTab.value = 'messages';
                };

                const updateMsg = async (msg) => {
                    await updateDoc(doc(db, 'artifacts', appId, 'public', 'data', 'messages', msg.id), { content: msg.content });
                };

                const deleteMsg = async (id) => {
                    await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'messages', id));
                };

                const generateCloud = () => {
                    const text = messages.value.map(m => m.content).join(' ');
                    // Simple word splitting for demo (Chinese splitting normally needs a library, here we simulate)
                    const words = text.split(/[\s,，。！!？?]/).filter(w => w.length > 1);
                    const freq = {};
                    words.forEach(w => freq[w] = (freq[w] || 0) + 1);
                    
                    const sorted = Object.entries(freq).sort((a,b) => b[1] - a[1]).slice(0, 20);
                    const colors = ['#60a5fa', '#34d399', '#f472b6', '#fbbf24', '#a78bfa', '#f87171'];
                    
                    cloudWords.value = sorted.map(([text, count]) => ({
                        text,
                        size: Math.min(40, 12 + count * 5),
                        color: colors[Math.floor(Math.random() * colors.length)],
                        x: 10 + Math.random() * 80,
                        y: 10 + Math.random() * 80,
                        opacity: 0.6 + (count/sorted[0][1]) * 0.4
                    }));
                };

                const updateChart = () => {
                    const ctx = document.getElementById('healthChart');
                    if (!ctx) return;
                    
                    const data = {
                        labels: ['安全留言', '已拦截违规'],
                        datasets: [{
                            data: [messages.value.length - violatedCount.value, violatedCount.value],
                            backgroundColor: ['rgba(59, 130, 246, 0.6)', 'rgba(239, 68, 68, 0.6)'],
                            borderColor: ['#3b82f6', '#ef4444'],
                            borderWidth: 2
                        }]
                    };

                    if (chartInstance) chartInstance.destroy();
                    chartInstance = new Chart(ctx, {
                        type: 'doughnut',
                        data: data,
                        options: {
                            responsive: true,
                            maintainAspectRatio: false,
                            plugins: {
                                legend: { position: 'bottom', labels: { color: '#94a3b8', boxWidth: 12, padding: 20 } }
                            },
                            cutout: '70%'
                        }
                    });
                };

                watch(currentTab, (newTab) => {
                    if (newTab === 'dashboard') {
                        setTimeout(() => {
                            generateCloud();
                            updateChart();
                        }, 100);
                    }
                });

                const formatTime = (ts) => ts ? new Date(ts.seconds * 1000).toLocaleString() : 'Just now';

                return {
                    currentTab, messages, loading, simulating, aiConfig, sim, tabName, lastSync,
                    violatedCount, activeUsersCount, cloudWords,
                    saveSettings, postSim, updateMsg, deleteMsg, generateCloud, formatTime
                };
            }
        }).mount('#app');
    </script>
