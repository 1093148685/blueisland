import React, { useState, useEffect, useRef } from 'react';
import { initializeApp } from 'firebase/app';
import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from 'firebase/auth';
import { getFirestore, collection, addDoc, query, onSnapshot, doc, serverTimestamp, updateDoc } from 'firebase/firestore';
import { Search, Send, Home, Lock, Music, Play, Pause, SkipForward, SkipBack, Sparkles, RefreshCw, ChevronUp, MessageCircleHeart, Ghost, Waves, CloudRain, Flame, Wind, Bot, X, Volume2 } from 'lucide-react';

/**
 * 环境配置
 */
const firebaseConfig = JSON.parse(__firebase_config);
const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getFirestore(app);
const appId = typeof __app_id !== 'undefined' ? __app_id : 'blue-island-v2-spirit';
const apiKey = ""; // 系统自动注入

const playlist = [
  { title: "青花瓷", artist: "周杰伦", duration: "3:31" },
  { title: "晴天", artist: "周杰伦", duration: "4:29" },
  { title: "兰亭序", artist: "周杰伦", duration: "4:13" },
  { title: "消愁", artist: "毛不易", duration: "4:18" },
  { title: "有何不可", artist: "许嵩", duration: "3:58" },
];

export default function App() {
  const [view, setView] = useState('home'); 
  const [user, setUser] = useState(null);
  const [messages, setMessages] = useState([]);
  const [isMusicOpen, setIsMusicOpen] = useState(false);
  const [currentSongIndex, setCurrentSongIndex] = useState(0);
  const [isPlaying, setIsPlaying] = useState(false);
  
  // 岛屿环境状态
  const [islandQuote, setIslandQuote] = useState("正在感应岛屿的气息...");
  const [islandMood, setIslandMood] = useState({ color: '#05162a', weather: '微风' });

  // 岛屿之灵状态
  const [isSpiritOpen, setIsSpiritOpen] = useState(false);
  const [spiritChat, setSpiritChat] = useState([]);
  const [spiritInput, setSpiritInput] = useState('');
  const [isSpiritTyping, setIsSpiritTyping] = useState(false);

  // 留言相关
  const [content, setContent] = useState('');
  const [avatarType, setAvatarType] = useState('anonymous'); 
  const [qqNumber, setQqNumber] = useState('');
  const [selectedAnonAvatar, setSelectedAnonAvatar] = useState(0);

  const anonAvatars = [
    "https://api.dicebear.com/7.x/adventurer/svg?seed=Felix",
    "https://api.dicebear.com/7.x/adventurer/svg?seed=Aneka",
    "https://api.dicebear.com/7.x/adventurer/svg?seed=Buddy",
    "https://api.dicebear.com/7.x/adventurer/svg?seed=Max"
  ];

  // 1. 登录与鉴权
  useEffect(() => {
    const initAuth = async () => {
      if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
        await signInWithCustomToken(auth, __initial_auth_token);
      } else {
        await signInAnonymously(auth);
      }
    };
    initAuth();
    const unsubscribe = onAuthStateChanged(auth, setUser);
    return () => unsubscribe();
  }, []);

  // 2. 实时监听留言
  useEffect(() => {
    if (!user) return;
    const q = collection(db, 'artifacts', appId, 'public', 'data', 'messages');
    const unsubscribe = onSnapshot(q, (snapshot) => {
      const msgs = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
      const sorted = msgs.sort((a, b) => (b.createdAt?.seconds || 0) - (a.createdAt?.seconds || 0));
      setMessages(sorted);
      if (sorted.length > 0) analyzeMood(sorted[0].content);
    }, (err) => console.error(err));
    return () => unsubscribe();
  }, [user]);

  // AI 情绪分析
  const analyzeMood = async (txt) => {
    const res = await callGemini(`分析这段话并返回今日岛屿的JSON {"quote": "寄语", "color": "十六进制深色", "weather": "天气"}: ${txt}`, "你是一个感知岛屿灵魂的AI。");
    if (res) {
      try {
        const data = JSON.parse(res.replace(/```json|```/g, ''));
        setIslandQuote(data.quote);
        setIslandMood({ color: data.color, weather: data.weather });
      } catch (e) {}
    }
  };

  const callGemini = async (prompt, system) => {
    try {
      const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          contents: [{ parts: [{ text: prompt }] }],
          systemInstruction: { parts: [{ text: system }] }
        })
      });
      const result = await response.json();
      return result.candidates?.[0]?.content?.parts?.[0]?.text;
    } catch (e) { return null; }
  };

  // 岛屿之灵对话
  const handleSpiritTalk = async () => {
    if (!spiritInput.trim()) return;
    const msg = spiritInput;
    setSpiritChat(prev => [...prev, { role: 'user', text: msg }]);
    setSpiritInput('');
    setIsSpiritTyping(true);

    const reply = await callGemini(msg, "你是'岛屿之灵'，语气神秘、温柔、治愈。你的回答简短且富有文学气息。");
    setIsSpiritTyping(false);
    if (reply) setSpiritChat(prev => [...prev, { role: 'spirit', text: reply.trim() }]);
  };

  const handlePost = async () => {
    if (!content || !user) return;
    const avatar = avatarType === 'qq' ? `https://q1.qlogo.cn/g?b=qq&nk=${qqNumber}&s=100` : anonAvatars[selectedAnonAvatar];
    await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'messages'), {
      content, avatarUrl: avatar, author: avatarType === 'qq' ? qqNumber : '匿名岛民',
      createdAt: serverTimestamp(), likes: 0
    });
    setContent(''); setView('browse');
  };

  return (
    <div className="min-h-screen text-white relative transition-all duration-1000" style={{ backgroundColor: islandMood.color }}>
      <div className="fixed inset-0 z-0 bg-cover bg-center opacity-20" style={{ backgroundImage: 'url(https://images.unsplash.com/photo-1518837695005-2083093ee35b?q=80&w=2000)' }}></div>

      <div className="relative z-10 container mx-auto px-6 max-w-lg pb-32 pt-12">
        {view === 'home' && (
          <div className="flex flex-col items-center justify-center min-h-[70vh] space-y-12">
            <div className="relative">
              <div className="w-32 h-32 bg-blue-400/10 rounded-full flex items-center justify-center backdrop-blur-3xl animate-pulse">
                <Waves className="text-blue-300 animate-bounce" size={48} />
              </div>
              <div className="absolute -top-2 -right-2 bg-blue-500/40 px-3 py-1 rounded-full text-[10px] border border-white/20">
                {islandMood.weather}
              </div>
            </div>

            <div className="text-center">
              <h1 className="text-3xl font-serif tracking-[0.6em] mb-4">留言岛</h1>
              <p className="text-xs text-blue-100/40 italic font-serif px-6">{islandQuote}</p>
            </div>

            <button 
              onClick={() => setIsSpiritOpen(true)}
              className="group flex items-center gap-4 px-10 py-5 bg-white/5 backdrop-blur-2xl rounded-full border border-white/10 hover:bg-blue-600/20 transition-all"
            >
              <Bot className="text-blue-400" />
              <span className="text-xs tracking-[0.4em] font-bold">对话岛屿之灵</span>
              <div className="w-2 h-2 bg-blue-400 rounded-full animate-ping"></div>
            </button>

            <div className="w-full space-y-4">
              <button onClick={() => setView('send')} className="w-full py-5 bg-blue-600 rounded-3xl font-bold tracking-[0.4em] shadow-2xl">投递心事</button>
              <button onClick={() => setView('browse')} className="w-full py-5 bg-white/5 rounded-3xl font-bold tracking-[0.4em] border border-white/10">翻阅回忆</button>
            </div>
          </div>
        )}

        {/* 浏览与发送视图省略，保持与之前一致逻辑 */}
        {view === 'browse' && (
          <div className="animate-in fade-in duration-500">
            <button onClick={() => setView('home')} className="mb-8 opacity-40 text-xs flex items-center gap-2"><Home size={14}/>返回主页</button>
            {messages.map(msg => (
              <div key={msg.id} className="bg-white/5 backdrop-blur-xl p-6 rounded-[2rem] border border-white/5 mb-6">
                <div className="flex gap-4 items-center mb-4">
                  <img src={msg.avatarUrl} className="w-10 h-10 rounded-full border border-white/10" />
                  <span className="text-[10px] font-bold opacity-40">{msg.author}</span>
                </div>
                <p className="text-sm font-serif leading-relaxed text-white/80">{msg.content}</p>
              </div>
            ))}
          </div>
        )}

        {view === 'send' && (
          <div className="animate-in slide-in-from-bottom-8 duration-500">
            <button onClick={() => setView('home')} className="mb-8 opacity-40 text-xs">← 放弃抒写</button>
            <div className="bg-black/40 backdrop-blur-3xl p-8 rounded-[2.5rem] border border-white/10">
              <textarea 
                value={content} 
                onChange={e=>setContent(e.target.value)} 
                placeholder="在此刻写下你的灵魂絮语..." 
                className="w-full h-48 bg-transparent outline-none text-sm font-serif leading-relaxed" 
              />
              <button onClick={handlePost} className="w-full mt-8 py-5 bg-blue-600 rounded-3xl font-bold tracking-[0.5em]">投递入海</button>
            </div>
          </div>
        )}
      </div>

      {/* 岛屿之灵 对话浮层 */}
      <div className={`fixed inset-0 z-[100] transition-all duration-700 ${isSpiritOpen ? 'translate-y-0' : 'translate-y-full'}`}>
        <div className="absolute inset-0 bg-slate-950/95 backdrop-blur-3xl flex flex-col">
          <div className="p-8 flex justify-between items-center border-b border-white/5">
            <div className="flex items-center gap-3">
              <Sparkles className="text-blue-400 animate-pulse" />
              <span className="text-sm tracking-widest font-bold font-serif">之灵的耳语</span>
            </div>
            <button onClick={() => setIsSpiritOpen(false)} className="p-3 bg-white/5 rounded-full"><X size={24}/></button>
          </div>

          <div className="flex-1 overflow-y-auto p-8 space-y-8 custom-scrollbar">
            {spiritChat.map((msg, i) => (
              <div key={i} className={`flex ${msg.role === 'user' ? 'justify-end' : 'justify-start'}`}>
                <div className={`max-w-[85%] p-5 rounded-3xl text-xs leading-loose font-serif ${msg.role === 'user' ? 'bg-blue-600 shadow-xl' : 'bg-white/5 border border-white/10 italic'}`}>
                  {msg.text}
                </div>
              </div>
            ))}
            {isSpiritTyping && <div className="text-[10px] opacity-20 animate-pulse font-serif">岛屿正在编织回响...</div>}
          </div>

          <div className="p-8 pb-12">
            <div className="flex gap-4">
              <input 
                value={spiritInput}
                onChange={e => setSpiritInput(e.target.value)}
                onKeyDown={e => e.key === 'Enter' && handleSpiritTalk()}
                placeholder="对岛屿说点什么..."
                className="flex-1 bg-white/5 rounded-2xl px-6 text-xs outline-none border border-white/10 focus:border-blue-500/50"
              />
              <button onClick={handleSpiritTalk} className="w-14 h-14 bg-blue-600 rounded-2xl flex items-center justify-center shadow-2xl active:scale-90 transition-all"><Send size={20}/></button>
            </div>
          </div>
        </div>
      </div>

      <style>{`
        @import url('https://fonts.googleapis.com/css2?family=Noto+Serif+SC:wght@200;400;700&display=swap');
        .font-serif { font-family: 'Noto Serif SC', serif; }
        .custom-scrollbar::-webkit-scrollbar { width: 2px; }
        .custom-scrollbar::-webkit-scrollbar-thumb { background: rgba(255,255,255,0.1); }
      `}</style>
    </div>
  );
}
