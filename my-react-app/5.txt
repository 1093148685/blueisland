import React, { useState, useEffect, useRef, useMemo } from 'react';
import { initializeApp } from 'firebase/app';
import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from 'firebase/auth';
import { getFirestore, collection, addDoc, query, onSnapshot, doc, serverTimestamp, updateDoc } from 'firebase/firestore';
import { Search, Send, Home, Lock, Plus, Music, Play, Pause, SkipForward, SkipBack, Sparkles, RefreshCw, ChevronUp, MessageCircleHeart, Ghost, Waves, CloudRain, Flame, Wind } from 'lucide-react';

/**
 * FIREBASE & GEMINI CONFIG
 */
const firebaseConfig = JSON.parse(__firebase_config);
const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getFirestore(app);
const appId = typeof __app_id !== 'undefined' ? __app_id : 'blue-island-v2-evolved';
const apiKey = ""; 

// 模拟歌单数据
const playlist = [
  { title: "青花瓷", artist: "周杰伦", duration: "3:31" },
  { title: "晴天", artist: "周杰伦", duration: "4:29" },
  { title: "兰亭序", artist: "周杰伦", duration: "4:13" },
  { title: "消愁", artist: "毛不易", duration: "4:18" },
  { title: "有何不可", artist: "许嵩", duration: "3:58" },
];

export default function App() {
  const [view, setView] = useState('home'); 
  const [user, setUser] = useState(null);
  const [messages, setMessages] = useState([]);
  const [searchCode, setSearchCode] = useState('');
  const [isMusicOpen, setIsMusicOpen] = useState(false);
  const [currentSongIndex, setCurrentSongIndex] = useState(0);
  const [isPlaying, setIsPlaying] = useState(false);
  
  // 进化状态：AI 岛屿环境
  const [islandQuote, setIslandQuote] = useState("正在聆听岛屿的风声...");
  const [islandMood, setIslandMood] = useState({ color: 'rgba(5, 22, 42, 0.8)', weather: '微风', intensity: 1 });
  const [ambientSounds, setAmbientSounds] = useState({ waves: false, rain: false, fire: false });

  // 表单状态
  const [content, setContent] = useState('');
  const [secretCode, setSecretCode] = useState('');
  const [avatarType, setAvatarType] = useState('anonymous'); 
  const [qqNumber, setQqNumber] = useState('');
  const [selectedAnonAvatar, setSelectedAnonAvatar] = useState(0);
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [isAILoading, setIsAILoading] = useState(false);

  // 匿名头像池
  const anonAvatars = [
    "https://api.dicebear.com/7.x/adventurer/svg?seed=Felix",
    "https://api.dicebear.com/7.x/adventurer/svg?seed=Aneka",
    "https://api.dicebear.com/7.x/adventurer/svg?seed=Buddy",
    "https://api.dicebear.com/7.x/adventurer/svg?seed=Max",
    "https://api.dicebear.com/7.x/adventurer/svg?seed=Luna"
  ];

  // 1. 初始化鉴权 (RULE 3)
  useEffect(() => {
    const initAuth = async () => {
      if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
        await signInWithCustomToken(auth, __initial_auth_token);
      } else {
        await signInAnonymously(auth);
      }
    };
    initAuth();
    const unsubscribe = onAuthStateChanged(auth, setUser);
    return () => unsubscribe();
  }, []);

  // 2. 实时监听留言 (RULE 1 & 2)
  useEffect(() => {
    if (!user) return;
    const q = query(collection(db, 'artifacts', appId, 'public', 'data', 'messages'));
    const unsubscribe = onSnapshot(q, (snapshot) => {
      const msgs = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }));
      const sorted = msgs.sort((a, b) => (b.createdAt?.seconds || 0) - (a.createdAt?.seconds || 0));
      setMessages(sorted);
      
      // 每当留言更新，尝试分析岛屿情绪
      if (sorted.length > 0) {
        const recentContext = sorted.slice(0, 5).map(m => m.content).join(" ");
        analyzeIslandMood(recentContext);
      }
    }, (err) => console.error("Firestore Error:", err));
    return () => unsubscribe();
  }, [user]);

  // --- AI 核心进化逻辑 ---
  const callGemini = async (prompt, systemPrompt = "") => {
    try {
      const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          contents: [{ parts: [{ text: prompt }] }],
          systemInstruction: { parts: [{ text: systemPrompt }] }
        })
      });
      const result = await response.json();
      return result.candidates?.[0]?.content?.parts?.[0]?.text;
    } catch (e) { return null; }
  };

  const analyzeIslandMood = async (context) => {
    const prompt = `分析这些留言的情绪：${context}。请返回JSON：{"quote": "一句温柔的寄语", "color": "一个适合这种情绪的十六进制深蓝色调", "weather": "一种天气(如:细雨,星空,晚霞)", "intensity": 0.5}`;
    const res = await callGemini(prompt, "你是一个感知岛屿灵魂的AI。请仅返回JSON格式内容。");
    try {
      if (res) {
        const cleanRes = res.replace(/```json|```/g, '').trim();
        const data = JSON.parse(cleanRes);
        setIslandQuote(data.quote);
        setIslandMood({ color: data.color, weather: data.weather, intensity: data.intensity });
      }
    } catch (e) {
      console.log("AI Parsing Error");
    }
  };

  const handleAISmooth = async () => {
    if (!content) return;
    setIsAILoading(true);
    try {
      const res = await callGemini(`润色这段心事，增加文学的美感和深邃感，保留原意，100字内：${content}`, "你是一位忧郁且温柔的诗人。");
      if (res) setContent(res.trim());
    } finally { setIsAILoading(false); }
  };

  const handlePost = async () => {
    if (!content || !user) return;
    setIsSubmitting(true);
    try {
      const avatarUrl = avatarType === 'qq' && qqNumber
        ? `https://q1.qlogo.cn/g?b=qq&nk=${qqNumber}&s=100` 
        : anonAvatars[selectedAnonAvatar];
      
      const docRef = await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'messages'), {
        content,
        secretCode,
        avatarUrl,
        author: avatarType === 'qq' ? `旅人(${qqNumber})` : '匿名岛民',
        createdAt: serverTimestamp(),
        likes: 0,
        aiEcho: "✨ 岛屿正在聆听回响..."
      });

      // 异步生成 AI 回响
      callGemini(`给这段心事写一个极简且温柔的回响，15字以内：${content}`, "你是一个深夜的聆听者。").then(echo => {
        if (echo) updateDoc(doc(db, 'artifacts', appId, 'public', 'data', 'messages', docRef.id), { aiEcho: echo.trim() });
      });

      setContent(''); 
      setView('browse');
    } finally { setIsSubmitting(false); }
  };

  // --- 环境音效控制 ---
  const toggleAmbient = (type) => {
    setAmbientSounds(prev => ({ ...prev, [type]: !prev[type] }));
    // 实际项目中这里会控制音频标签
  };

  // 获取背景滤镜
  const getDynamicFilter = () => {
    const hour = new Date().getHours();
    if (hour >= 19 || hour < 5) return 'brightness(0.5) saturate(0.8) hue-rotate(20deg)'; // 深夜
    if (hour >= 5 && hour < 9) return 'brightness(0.8) sepia(0.2) hue-rotate(-10deg)'; // 清晨
    if (hour >= 16 && hour < 19) return 'brightness(0.7) sepia(0.5) hue-rotate(10deg)'; // 黄昏
    return 'none'; // 白天
  };

  return (
    <div className="min-h-screen text-white font-sans selection:bg-blue-500/30 relative overflow-x-hidden transition-colors duration-500" style={{ backgroundColor: islandMood.color }}>
      
      {/* 动态背景图层 */}
      <div 
        className="fixed inset-0 z-0 bg-cover bg-center transition-all duration-3000 scale-105" 
        style={{ 
          backgroundImage: 'url(https://images.unsplash.com/photo-1518837695005-2083093ee35b?q=80&w=2000)',
          filter: getDynamicFilter()
        }}
      >
        {/* 动态遮罩层 */}
        <div className="absolute inset-0 transition-colors duration-3000" style={{ backgroundColor: `${islandMood.color}cc` }}></div>
      </div>

      <div className="relative z-10 container mx-auto px-6 max-w-lg">
        {view === 'home' && (
          <div className="flex flex-col items-center justify-center min-h-[85vh] space-y-12">
            <div className="flex flex-col items-center pt-12 animate-in zoom-in duration-1000">
               <div className="relative">
                  <div className="w-28 h-28 bg-white/10 rounded-full flex items-center justify-center backdrop-blur-md shadow-2xl border border-white/20 animate-pulse">
                    <div className="text-5xl">🌑</div>
                  </div>
                  {/* 天气小图标 */}
                  <div className="absolute -top-2 -right-2 w-10 h-10 bg-blue-500/40 rounded-full flex items-center justify-center backdrop-blur-xl border border-white/30 text-[10px]">
                    {islandMood.weather === '细雨' ? <CloudRain size={16}/> : islandMood.weather === '星空' ? <Sparkles size={16}/> : <Wind size={16}/>}
                  </div>
               </div>
               <h2 className="mt-8 text-sm text-blue-100/90 tracking-[0.2em] font-serif text-center px-8 italic leading-relaxed min-h-[3rem]">
                {islandQuote}
               </h2>
               <div className="mt-2 text-[9px] opacity-40 uppercase tracking-widest">
                 岛屿当前：{islandMood.weather}
               </div>
            </div>

            <div className="w-full space-y-6">
              <button onClick={() => setView('send')} className="group relative w-full py-5 bg-blue-600/60 backdrop-blur-xl rounded-3xl font-bold tracking-[0.3em] hover:bg-blue-500 transition-all shadow-2xl border border-white/20 overflow-hidden">
                <span className="relative z-10">开启心灵信笺</span>
                <div className="absolute inset-0 bg-gradient-to-r from-transparent via-white/20 to-transparent -translate-x-full group-hover:translate-x-full transition-transform duration-1000"></div>
              </button>
              <button onClick={() => setView('browse')} className="w-full py-5 bg-white/5 backdrop-blur-md rounded-3xl font-bold tracking-[0.3em] hover:bg-white/10 transition-all shadow-xl border border-white/10">
                翻阅往昔心事
              </button>
            </div>
          </div>
        )}

        {view === 'send' && (
          <div className="pt-8 pb-24 animate-in slide-in-from-right-8 duration-500">
             <button onClick={() => setView('home')} className="mb-6 opacity-40 flex items-center text-xs hover:opacity-100 transition-all"><Home size={14} className="mr-2"/>回到中心</button>
             
             <div className="bg-black/40 backdrop-blur-2xl rounded-[2.5rem] p-8 border border-white/10 shadow-2xl space-y-8">
                <div>
                  <div className="flex justify-between mb-4">
                    <label className="text-[10px] opacity-40 tracking-[0.2em] font-bold">MESSAGE / 抒写</label>
                    <button onClick={handleAISmooth} disabled={isAILoading || !content} className="text-[10px] text-blue-300 flex items-center gap-1.5 hover:text-blue-100 transition-all bg-white/5 px-3 py-1 rounded-full border border-white/10">
                      {isAILoading ? <RefreshCw className="animate-spin" size={12}/> : <Sparkles size={12}/>} 诗人润色
                    </button>
                  </div>
                  <textarea 
                    value={content}
                    onChange={e => setContent(e.target.value)}
                    placeholder="此刻，让文字在寂静中流淌..."
                    className="w-full h-48 bg-white/5 rounded-3xl p-6 text-sm focus:ring-1 focus:ring-blue-400 outline-none border border-white/5 resize-none placeholder:text-white/10 transition-all leading-relaxed"
                  />
                </div>

                <div className="grid grid-cols-2 gap-6">
                  <div>
                    <label className="text-[10px] opacity-40 block mb-3 tracking-[0.2em] font-bold">PASSCODE / 暗语</label>
                    <input 
                      type="text"
                      value={secretCode}
                      onChange={e => setSecretCode(e.target.value)}
                      placeholder="检索凭证..."
                      className="w-full bg-white/5 rounded-2xl p-4 text-xs outline-none border border-white/5 focus:ring-1 focus:ring-blue-400"
                    />
                  </div>
                  <div>
                    <label className="text-[10px] opacity-40 block mb-3 tracking-[0.2em] font-bold">AVATAR / 身份</label>
                    <div className="flex bg-white/5 p-1.5 rounded-2xl border border-white/5">
                      <button onClick={() => setAvatarType('anonymous')} className={`flex-1 py-2 text-[10px] rounded-xl transition-all ${avatarType === 'anonymous' ? 'bg-blue-600 shadow-lg' : 'opacity-30'}`}>匿名</button>
                      <button onClick={() => setAvatarType('qq')} className={`flex-1 py-2 text-[10px] rounded-xl transition-all ${avatarType === 'qq' ? 'bg-blue-600 shadow-lg' : 'opacity-30'}`}>QQ号</button>
                    </div>
                  </div>
                </div>

                {avatarType === 'anonymous' ? (
                  <div className="flex justify-between px-2 py-2">
                    {anonAvatars.map((url, i) => (
                      <button key={i} onClick={() => setSelectedAnonAvatar(i)} className={`w-11 h-11 rounded-2xl overflow-hidden border-2 transition-all ${selectedAnonAvatar === i ? 'border-blue-400 scale-110 shadow-2xl' : 'border-transparent opacity-20 hover:opacity-100'}`}>
                        <img src={url} alt="avatar" />
                      </button>
                    ))}
                  </div>
                ) : (
                  <input 
                    type="text"
                    value={qqNumber}
                    onChange={e => setQqNumber(e.target.value)}
                    placeholder="输入QQ号自动同步头像..."
                    className="w-full bg-white/5 rounded-2xl p-4 text-xs border border-white/5 outline-none focus:ring-1 focus:ring-blue-400"
                  />
                )}

                <button 
                  onClick={handlePost}
                  disabled={isSubmitting || !content}
                  className="w-full py-5 bg-gradient-to-br from-blue-600 to-indigo-700 rounded-3xl font-bold tracking-[0.4em] shadow-2xl active:scale-95 transition-all disabled:opacity-20 flex items-center justify-center gap-3"
                >
                  {isSubmitting ? <RefreshCw className="animate-spin" size={18}/> : null}
                  {isSubmitting ? '封印思绪中...' : '投递入海'}
                </button>
             </div>
          </div>
        )}

        {view === 'browse' && (
          <div className="pt-8 pb-24 animate-in fade-in duration-700">
            <div className="sticky top-6 z-30 mb-10">
              <div className="relative">
                <input 
                  type="text"
                  value={searchCode}
                  onChange={e => setSearchCode(e.target.value)}
                  placeholder="寻找那句暗语..."
                  className="w-full pl-8 pr-16 py-5 bg-white/10 backdrop-blur-3xl border border-white/20 rounded-[2rem] outline-none focus:ring-2 focus:ring-blue-400 transition-all shadow-2xl placeholder:text-white/20 text-sm"
                />
                <div className="absolute right-5 top-1/2 -translate-y-1/2 w-10 h-10 bg-blue-500/60 rounded-full flex items-center justify-center shadow-lg">
                  <Search size={18} />
                </div>
              </div>
            </div>

            <div className="space-y-12 pl-4 relative">
              <div className="absolute left-1 top-0 bottom-0 w-[1px] bg-gradient-to-b from-blue-400/60 via-blue-400/10 to-transparent"></div>
              
              {(searchCode ? messages.filter(m => m.secretCode === searchCode) : messages).map((msg, i) => (
                <div key={msg.id} className="relative animate-in slide-in-from-left-4" style={{ animationDelay: `${i * 80}ms` }}>
                  <div className="absolute -left-[16px] top-6 w-2 h-2 rounded-full bg-white shadow-[0_0_15px_rgba(255,255,255,1)]"></div>
                  
                  <div className="bg-white/5 backdrop-blur-xl rounded-[2rem] p-8 border border-white/10 shadow-xl group hover:bg-white/10 transition-all">
                    <div className="flex items-center gap-4 mb-5">
                      <img src={msg.avatarUrl} alt="" className="w-12 h-12 rounded-2xl border border-white/10 object-cover shadow-xl transition-transform group-hover:scale-110" />
                      <div>
                        <div className="text-sm font-bold text-blue-100/90 tracking-wide">{msg.author}</div>
                        <div className="text-[10px] opacity-30 mt-0.5">
                          {msg.createdAt?.seconds ? new Date(msg.createdAt.seconds * 1000).toLocaleString() : '刚刚抵达'}
                        </div>
                      </div>
                    </div>
                    <p className="text-base leading-relaxed text-white/80 whitespace-pre-wrap font-serif italic mb-6 tracking-wide">“ {msg.content} ”</p>
                    
                    {msg.aiEcho && (
                      <div className="mt-6 p-4 bg-blue-500/10 rounded-2xl border border-blue-400/20 flex gap-3 items-start animate-in fade-in duration-1000">
                        <MessageCircleHeart className="text-blue-400 shrink-0 mt-1 animate-pulse" size={14} />
                        <div className="text-xs text-blue-200/70 italic font-serif leading-relaxed">
                          {msg.aiEcho}
                        </div>
                      </div>
                    )}

                    <div className="mt-6 flex justify-between items-center">
                       {msg.secretCode ? (
                         <div className="inline-flex items-center px-3 py-1 rounded-full bg-white/5 text-[9px] text-white/30 border border-white/5 uppercase tracking-widest">
                            <Lock size={10} className="mr-2"/> {msg.secretCode}
                         </div>
                       ) : <div/>}
                       
                       <button 
                        onClick={() => {
                          const ref = doc(db, 'artifacts', appId, 'public', 'data', 'messages', msg.id);
                          updateDoc(ref, { likes: (msg.likes || 0) + 1 });
                        }}
                        className="flex items-center gap-2 px-4 py-2 rounded-full bg-white/5 hover:bg-blue-500/20 transition-all group/like"
                       >
                         <Sparkles size={12} className={msg.likes > 0 ? "text-blue-400" : "opacity-30"} />
                         <span className="text-[10px] font-bold opacity-40">{msg.likes || 0}</span>
                       </button>
                    </div>
                  </div>
                </div>
              ))}

              {messages.length === 0 && (
                <div className="text-center py-32 opacity-10">
                  <Ghost size={64} className="mx-auto mb-6" />
                  <div className="text-sm font-serif italic tracking-[0.5em]">寂静之岛</div>
                </div>
              )}
            </div>
          </div>
        )}
      </div>

      {/* 底部导航进化 */}
      <div className="fixed bottom-0 left-0 right-0 z-40 px-6 py-8 bg-gradient-to-t from-black/80 to-transparent pointer-events-none">
        <div className="max-w-lg mx-auto flex justify-between items-center pointer-events-auto">
          <button onClick={() => setView('home')} className={`w-14 h-14 rounded-2xl flex items-center justify-center transition-all ${view === 'home' ? 'bg-blue-600 shadow-[0_0_20px_rgba(37,99,235,0.4)] scale-110' : 'bg-white/5 opacity-30 hover:opacity-100'}`}>
            <Home size={24}/>
          </button>
          
          <div className="flex-1 mx-6 bg-white/5 backdrop-blur-2xl border border-white/10 h-14 rounded-2xl flex items-center justify-center text-[10px] tracking-[0.4em] font-serif text-white/40 shadow-inner">
             {view === 'home' ? 'ISLAND' : view === 'send' ? 'WRITE' : 'WHISPER'}
          </div>

          <button onClick={() => setView('browse')} className={`w-14 h-14 rounded-2xl flex items-center justify-center transition-all ${view === 'browse' ? 'bg-blue-600 shadow-[0_0_20px_rgba(37,99,235,0.4)] scale-110' : 'bg-white/5 opacity-30 hover:opacity-100'}`}>
            <Search size={24}/>
          </button>
        </div>
      </div>

      {/* 音乐与环境音控制中心 */}
      <div className={`fixed bottom-28 right-6 z-50 transition-all duration-700 ${isMusicOpen ? 'w-72 opacity-100 translate-y-0 scale-100' : 'w-14 opacity-0 translate-y-10 scale-90 pointer-events-none'}`}>
        <div className="bg-slate-900/90 backdrop-blur-3xl rounded-[2rem] border border-white/10 shadow-2xl overflow-hidden pointer-events-auto">
          {/* 顶部歌曲信息 */}
          <div className="p-5 flex items-center gap-4 bg-white/5 border-b border-white/5">
             <div className={`w-12 h-12 bg-blue-600 rounded-full flex items-center justify-center shadow-2xl ${isPlaying ? 'animate-spin-slow' : ''}`}>
                <Music size={20}/>
             </div>
             <div className="overflow-hidden flex-1">
                <div className="text-xs font-bold truncate text-white">{playlist[currentSongIndex].title}</div>
                <div className="text-[10px] opacity-40 truncate mt-1">{playlist[currentSongIndex].artist}</div>
             </div>
          </div>
          
          <div className="p-6 space-y-6">
            {/* 播放控制 */}
            <div className="flex items-center justify-center gap-8">
              <button onClick={() => setCurrentSongIndex((currentSongIndex - 1 + playlist.length) % playlist.length)} className="opacity-30 hover:opacity-100 transition-all"><SkipBack size={20}/></button>
              <button onClick={() => setIsPlaying(!isPlaying)} className="w-12 h-12 bg-white text-slate-900 rounded-full flex items-center justify-center shadow-xl active:scale-90 transition-all">
                {isPlaying ? <Pause size={24}/> : <Play size={24} className="ml-1"/>}
              </button>
              <button onClick={() => setCurrentSongIndex((currentSongIndex + 1) % playlist.length)} className="opacity-30 hover:opacity-100 transition-all"><SkipForward size={20}/></button>
            </div>

            {/* 环境音 ASMR 叠播控制 */}
            <div className="space-y-3">
               <div className="text-[9px] opacity-20 uppercase tracking-[0.2em] font-bold">ASMR 环境叠播</div>
               <div className="flex justify-between gap-2">
                  <button onClick={() => toggleAmbient('waves')} className={`flex-1 py-3 rounded-xl border border-white/5 flex flex-col items-center gap-1.5 transition-all ${ambientSounds.waves ? 'bg-blue-600/40 text-blue-300' : 'bg-white/5 opacity-40 hover:opacity-100'}`}>
                    <Waves size={16}/> <span className="text-[8px]">海浪</span>
                  </button>
                  <button onClick={() => toggleAmbient('rain')} className={`flex-1 py-3 rounded-xl border border-white/5 flex flex-col items-center gap-1.5 transition-all ${ambientSounds.rain ? 'bg-blue-600/40 text-blue-300' : 'bg-white/5 opacity-40 hover:opacity-100'}`}>
                    <CloudRain size={16}/> <span className="text-[8px]">细雨</span>
                  </button>
                  <button onClick={() => toggleAmbient('fire')} className={`flex-1 py-3 rounded-xl border border-white/5 flex flex-col items-center gap-1.5 transition-all ${ambientSounds.fire ? 'bg-blue-600/40 text-blue-300' : 'bg-white/5 opacity-40 hover:opacity-100'}`}>
                    <Flame size={16}/> <span className="text-[8px]">篝火</span>
                  </button>
               </div>
            </div>
            
            <div className="space-y-2 max-h-32 overflow-y-auto pr-2 custom-scrollbar border-t border-white/5 pt-4">
               {playlist.map((song, i) => (
                 <div key={i} onClick={() => setCurrentSongIndex(i)} className={`text-[10px] p-2.5 rounded-xl cursor-pointer transition-all ${currentSongIndex === i ? 'bg-blue-600/30 text-blue-300' : 'hover:bg-white/5 opacity-30'}`}>
                   <div className="flex justify-between items-center">
                     <span className="font-medium">{song.title}</span>
                     <span className="opacity-40">{song.duration}</span>
                   </div>
                 </div>
               ))}
            </div>
          </div>
        </div>
      </div>

      {/* 悬浮球控制 */}
      <button 
        onClick={() => setIsMusicOpen(!isMusicOpen)}
        className="fixed bottom-28 right-6 w-14 h-14 bg-white/10 backdrop-blur-2xl rounded-2xl border border-white/10 flex items-center justify-center z-50 hover:bg-blue-600/40 transition-all shadow-2xl group"
      >
        <Music size={22} className={`transition-all ${isPlaying ? 'animate-bounce text-blue-400' : 'opacity-30'}`} />
        {(isPlaying || Object.values(ambientSounds).some(v => v)) && (
          <div className="absolute -top-1 -right-1 w-4 h-4 bg-blue-500 rounded-full border-2 border-slate-900 animate-ping"></div>
        )}
      </button>

      {/* 回到顶部 */}
      <button 
        onClick={() => window.scrollTo({ top: 0, behavior: 'smooth' })}
        className="fixed bottom-44 right-6 w-14 h-14 bg-white/5 backdrop-blur-md rounded-full border border-white/10 flex items-center justify-center z-40 hover:bg-white/10 transition-all opacity-20 hover:opacity-100"
      >
        <ChevronUp size={22} />
      </button>

      <style>{`
        @import url('https://fonts.googleapis.com/css2?family=Noto+Serif+SC:wght@200;400;700&display=swap');
        .font-serif { font-family: 'Noto Serif SC', serif; }
        .animate-spin-slow { animation: spin 12s linear infinite; }
        @keyframes spin { from { transform: rotate(0deg); } to { transform: rotate(360deg); } }
        .custom-scrollbar::-webkit-scrollbar { width: 3px; }
        .custom-scrollbar::-webkit-scrollbar-track { background: transparent; }
        .custom-scrollbar::-webkit-scrollbar-thumb { background: rgba(255,255,255,0.05); border-radius: 10px; }
        .transition-colors { transition: background-color 3s ease, color 0.5s ease; }
        .transition-all { transition: all 0.5s cubic-bezier(0.4, 0, 0.2, 1); }
      `}</style>
    </div>
  );
}
