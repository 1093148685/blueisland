    <script type="module">
        import { initializeApp } from 'https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js';
        import { getAuth, signInAnonymously, onAuthStateChanged, signInWithCustomToken } from 'https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js';
        import { getFirestore, collection, query, onSnapshot, doc, deleteDoc, setDoc, getDoc } from 'https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js';

        const { createApp, ref, computed, onMounted } = Vue;

        const firebaseConfig = JSON.parse(__firebase_config);
        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'blue-island-v2-reborn';

        createApp({
            setup() {
                const currentTab = ref('stats');
                const user = ref(null);
                const messages = ref([]);
                const filterText = ref('');
                const deletingId = ref(null);
                const saveStatus = ref('');

                // AI 配置状态
                const aiConfig = ref({
                    systemPrompt: '你是一个细腻温暖、充满同情心的岛屿守护灵。面对人们投递的留言，请给予治愈、轻盈且带有文学美感的回响，回复请保持在100字以内。',
                    temperature: 0.7,
                    maxTokens: 200,
                    autoReply: true
                });

                const tabTitle = computed(() => {
                    const titles = { stats: '岛屿脉动', messages: '思绪审阅', ai: '智能核心' };
                    return titles[currentTab.value];
                });

                onMounted(async () => {
                    const initAuth = async () => {
                        if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                            await signInWithCustomToken(auth, __initial_auth_token);
                        } else {
                            await signInAnonymously(auth);
                        }
                    };
                    await initAuth();
                    onAuthStateChanged(auth, u => {
                        user.value = u;
                        if (u) {
                            listenMessages();
                            loadAIConfig();
                        }
                    });
                });

                const listenMessages = () => {
                    const q = query(collection(db, 'artifacts', appId, 'public', 'data', 'messages'));
                    onSnapshot(q, (snapshot) => {
                        messages.value = snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() }))
                            .sort((a, b) => (b.createdAt?.seconds || 0) - (a.createdAt?.seconds || 0));
                    }, (err) => console.error("Message Snapshot error:", err));
                };

                const loadAIConfig = async () => {
                    // 修复：确保文档路径为偶数段 (6段)
                    // 原路径: artifacts/{appId}/public/data/aiSettings (5段 - 报错)
                    // 新路径: artifacts/{appId}/public/data/settings/aiConfig (6段 - 正确)
                    try {
                        const configRef = doc(db, 'artifacts', appId, 'public', 'data', 'settings', 'aiConfig');
                        const snap = await getDoc(configRef);
                        if (snap.exists()) {
                            aiConfig.value = { ...aiConfig.value, ...snap.data() };
                        }
                    } catch (e) {
                        console.error("Load Config Error:", e);
                    }
                };

                const saveAIConfig = async () => {
                    if (!user.value) return;
                    saveStatus.value = '正在上传配置...';
                    try {
                        const configRef = doc(db, 'artifacts', appId, 'public', 'data', 'settings', 'aiConfig');
                        await setDoc(configRef, aiConfig.value);
                        saveStatus.value = '配置已同步';
                        setTimeout(() => saveStatus.value = '', 3000);
                    } catch (e) {
                        saveStatus.value = '保存失败';
                        console.error("Save Config Error:", e);
                    }
                };

                const totalLikes = computed(() => messages.value.reduce((acc, curr) => acc + (curr.likes || 0), 0));
                const aiEchoCount = computed(() => messages.value.filter(m => m.echo || m.mood).length);
                const filteredMessages = computed(() => {
                    if (!filterText.value) return messages.value;
                    const ft = filterText.value.toLowerCase();
                    return messages.value.filter(m => m.content.toLowerCase().includes(ft) || m.author.toLowerCase().includes(ft));
                });

                const confirmDelete = (id) => { deletingId.value = id; };
                const handleDelete = async () => {
                    if (!deletingId.value) return;
                    await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', 'messages', deletingId.value));
                    deletingId.value = null;
                };

                const formatTime = (ts) => ts ? new Date(ts.seconds * 1000).toLocaleString() : '...';

                return {
                    currentTab, tabTitle, user, messages, filterText, deletingId,
                    aiConfig, saveAIConfig, saveStatus,
                    totalLikes, aiEchoCount, filteredMessages,
                    confirmDelete, handleDelete, formatTime
                };
            }
        }).mount('#app');
    </script>
